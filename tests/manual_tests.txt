## PATCH
curl -H "Content-type: application/json" -X PATCH http://127.0.0.1:5000/api/v1.0/team/0 -d '{"email":"elmer.thomas@sendgrid.com"}'

## PUT
curl -X PUT \
     -H "Content-Type: application/json" \
     -d '{
    "type": "Ambassador",
    "first_name": "Ultimate",
    "last_name": "Warrior",
    "email": "ultimate@warrior.com",
    "phone": "999.999.9999",
    "home_city": "Parts Unknown"
}' \
http://127.0.0.1:5000/api/v1.0/team

## DELETE
curl --include --request DELETE http://127.0.0.1:5000/api/v1.0/team/0
